#!/bin/bash

list1=`seq -w 07 38`
n=0
home=$PWD
for i in ${list1[*]}
do
  mkdir $i
  list2=`seq -w 01 10`
  for j in ${list2[*]}
  do
    mkdir $i/$j
    echo $i $j
    cd $home
  done
done
